<goax-sidebar (_change)="handleSelectMenuItem($event)">

      <goax-sidebar-item
      [active]="subMenu.active"
      [name]="subMenu.name"
      [displayname]="subMenu.displayname"
      [url]="subMenu.url">
     </goax-sidebar-item>
   </ng-template>
  </goax-sidebar-item>
</goax-sidebar>
